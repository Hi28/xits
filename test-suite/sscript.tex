\def\TestBoth#1#2#3{
    \mathematics{#1#2#3%
        \utfchar\One^{\utfchar\Two}\kern5pt
        \utfchar\One_{\utfchar\Two}\kern5pt
        \utfchar\One^{\utfchar\Two}_{\utfchar\Two}
    }
    \hskip10pt
}
